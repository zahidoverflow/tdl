# TDL installer bootstrap # One-liner: irm https://zahidoverflow.github.io/tdl | iex $ErrorActionPreference = 'Stop' $installerUrl = 'https://raw.githubusercontent.com/zahidoverflow/tdl/master/install.ps1' $script = Invoke-RestMethod -Uri $installerUrl -Headers @{ 'Cache-Control' = 'no-cache' } if ([string]::IsNullOrWhiteSpace($script)) { throw \"Failed to download installer from $installerUrl\" } Invoke-Expression $script
